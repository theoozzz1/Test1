\subsection*{Introduction}

This API documentation describes the Byte Me backend REST API, generated via \textbf{Swagger}. It explains how to call each endpoint using \texttt{curl}, along with request and response examples.

Including these seven controllers:


\begin{enumerate}
    \setcounter{enumi}{1}
    \item \textbf{auth-controller}
    \item \textbf{bundle-controller}
    \item \textbf{category-controller}
    \item \textbf{order-controller}
    \item \textbf{issue-controller}
    \item \textbf{gamification-controller}
    \item \textbf{AnalyticsController}
\end{enumerate}






\section{ AuthController (\texttt{/api/auth})}

\subsection{Register}
\textbf{Purpose:} \\ \texttt Register a new user account and create a profile for a specific role. \\

\textbf{POST} \texttt{/api/auth/register} \\

\subsubsection*{Example Request}
\begin{lstlisting}
curl -X POST http://localhost:8080/api/auth/register \
  -H "Content-Type: application/json" \
  -d '{
    "email": "seller@example.com",
    "password": "password123",
    "role": "SELLER",
    "businessName": "ABC",
    "location": "Exeter"
  }'
\end{lstlisting}

\subsubsection*{Example Response}
\begin{lstlisting}
{
  "token": "string",
  "userId": "3fa85f64-5717-4562-b3fc-2c963f66afa6",
  "profileId": "3fa85f64-5717-4562-b3fc-2c963f66afa6",
  "email": "string",
  "role": "SELLER"
}
\end{lstlisting}
